27""" Module to centralize command line argument parsing """ 

28import argparse 

29from enum import Enum 

30from typing import List 

31from tuna.helper import ConfigType 

32 

33 

34class TunaArgs(Enum): 

35 """ Enumeration of all the common argument supported by setup_arg_parser """ 

36 ARCH = 0 

37 NUM_CU = 1 

38 DIRECTION = 2 

39 VERSION = 3 

40 CONFIG_TYPE = 4 

41 

42 

43def setup_arg_parser(desc: str, arg_list: List[TunaArgs]): 

44 """ function to aggregate common command line args """ 

45 parser = argparse.ArgumentParser(description=desc) 

46 if TunaArgs.ARCH in arg_list: 

47 parser.add_argument( 

48 '-a', 

49 '--arch', 

50 type=str, 

51 dest='arch', 

52 default=None, 

53 required=False, 

54 help='Architecture of machines', 

55 choices=['gfx900', 'gfx906', 'gfx908', 'gfx1030', 'gfx90a']) 

56 if TunaArgs.NUM_CU in arg_list: 

57 parser.add_argument('-n', 

58 '--num_cu', 

59 dest='num_cu', 

60 type=int, 

61 default=None, 

62 required=False, 

63 help='Number of CUs on GPU', 

64 choices=[36, 56, 60, 64, 104, 110, 120]) 

65 if TunaArgs.DIRECTION in arg_list: 

66 parser.add_argument( 

67 '-d', 

68 '--direction', 

69 type=str, 

70 dest='direction', 

71 default=None, 

72 help='Direction of tunning, None means all (fwd, bwd, wrw), \ 

73 fwd = 1, bwd = 2, wrw = 4.', 

74 choices=[None, '1', '2', '4']) 

75 if TunaArgs.CONFIG_TYPE in arg_list: 

76 parser.add_argument('-C', 

77 '--config_type', 

78 dest='config_type', 

79 help='Specify configuration type', 

80 default=ConfigType.convolution, 

81 choices=ConfigType, 

82 type=ConfigType) 

83 return parser
